import turtle

# --- Configuracao Padrao ---
screen = turtle.Screen ()
t = turtle.Turtle ()
screen.title (" Resultado - Exemplo 1")
# --- Codigo Gerado pelo Compilador ---
t.forward (150)
t.right (90)
t.forward (150)
t.right (90)
t.forward (150)
t.right (90)
t.forward (150)
t.right (90)

# --- Finalizacao ---
turtle.done ()
